#include <iostream>
#include "windows.h"

#define _USE_MATH_DEFINES 
#include <math.h> 
#include <cmath>

using namespace std;

int main()
{
    SetConsoleCP(1251);
    SetConsoleOutputCP(1251);

    double sx = 345343434;
    double ex = 50000000;
    double h = 12345678910;

    double x = sx;

    cout << "x\t| Y\n";
    

    while (x <= ex)
    {
        double Y = pow(x, 3) + log(fabs(x)) - 10000000000000003;
        cout << x << "\t| " << Y << endl;
        x += h;
    }



    return 0;
}
